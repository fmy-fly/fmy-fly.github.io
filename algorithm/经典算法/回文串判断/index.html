    bool checkPartitioning(string s) {
        int n = s.size();
        bool d[n ][n ];
        memset(d,false,sizeof(d));
        for (int i = 0;i < n; i++){
            for (int j = i; j >= 0; j--){
                if ( i == j) d[j][i] = true;
                else{
                    if (s[i] == s[j]){
                        if (i - j == 1 || i - j == 2 || d[j + 1][i - 1]) d[j][i] = true;
                    }
                }
            }
        }



}
